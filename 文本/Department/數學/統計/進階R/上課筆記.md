hr_leader <- function(data) {
  data |> 
    group_by(playerID) |> 
    summarize(HR = sum(HR)) |>
    arrange(desc(HR)) |>
    slice(1)
}

  agg <- aggregate(HR ~ playerID, data = df, sum, na.rm = TRUE)
  agg[which.max(agg$HR), , drop = FALSE]
}
